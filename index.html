extension StringExtensions on String {

    if (isEmpty) return this;
    return this[0].toUpperCase() + substring(1);
  }
  
  bool get isEmail => contains('@');
  
  String reverse() {
    return split('').reversed.join();
  }
}

extension DateTimeExtensions on DateTime {
  // Method
  String toFormattedString() {
    return '$day/$month/$year';
  }
  
  // Getter
  bool get isToday {
    var now = DateTime.now();
    return year == now.year && 
           month == now.month && 
           day == now.day;
  }
  
  // Operator
  DateTime operator +(Duration duration) {
    return add(duration);
  }
}

void main() {
  // 1. Implementasi toFormattedString()
  print('--- toFormattedString() ---');
  final now = DateTime.now();
  print('Current date: ${now.toFormattedString()}');

  final specificDate = DateTime(2023, 10, 26);
  print('Specific date (2023-10-26): ${specificDate.toFormattedString()}');

  print('\n--- isToday ---');
  // 2. Implementasi isToday
  final today = DateTime.now();
  final tomorrow = DateTime.now().add(Duration(days: 1));
  final yesterday = DateTime.now().subtract(Duration(days: 1));

  print('Is ${today.toFormattedString()} today? ${today.isToday}');
  print('Is ${tomorrow.toFormattedString()} today? ${tomorrow.isToday}');
  print('Is ${yesterday.toFormattedString()} today? ${yesterday.isToday}');

  print('\n--- operator + ---');
  // 3. Implementasi operator +
  final initialDate = DateTime(2024, 1, 15);
  final durationToAdd = Duration(days: 10, hours: 3);
  final futureDate = initialDate + durationToAdd;

  print('Initial date: ${initialDate.toFormattedString()}');
  print('Duration to add: ${durationToAdd.inDays} days, ${durationToAdd.inHours % 24} hours');
  print('Future date: ${futureDate.toFormattedString()}');
  print('Future date (full): $futureDate');

  final anotherDuration = Duration(days: 2);

extension NumberExtensions on num {
  // Format ke Rupiah
  String toRupiah() {
    return 'Rp${toStringAsFixed(0).replaceAllMapped(
      RegExp(r'(\d{1,3})(?=(\d{3})+(?!\d))'),
      (match) => '${match[1]}.',
    )}';
  }
  
  // Check jika dalam range
  bool isBetween(num min, num max) {
    return this >= min && this <= max;
  }
  
  // Pangkat
  num power(int exponent) {
    var result = this;
    for (var i = 1; i < exponent; i++) {
      result *= this;
    }
    return result;
  }
}

extension ListExtensions on List {
  // Safe get dengan default
  T? getOrNull(int index) {
    if (index < 0 || index >= length) return null;
    return this[index];
  }
  
  // Split list menjadi chunks
  List> chunk(int size) {
    var chunks = >[];
    for (var i = 0; i < length; i += size) {
      chunks.add(sublist(i, i + size > length ? length : i + size));
    }
    return chunks;
  }
  
  // Distinct items
  List distinct() {
    var result = [];
    for (var item in this) {
      if (!result.contains(item)) {
        result.add(item);
      }
    }
    return result;
  }
}

void main(){
// Penggunaan

print(numbers.chunk(3));  // [[1, 2, 3], [4, 5, 6], [7]]
print([1, 2, 2, 3, 3, 3].distinct()); // [1, 2, 3]

class Product {
  final String name;
  final double price;
  
  Product(this.name, this.price);

// Extension untuk formatting double ke Rupiah
extension DoubleRupiahFormatter on double {
  String toRupiah() {
    // Convert to integer to handle whole Rupiah amounts without decimals
    final intValue = toInt();
    
    // Convert to string
    String numberString = intValue.toString();
    
    // Add thousands separators
    String formattedString = '';
    int counter = 0;
    for (int i = numberString.length - 1; i >= 0; i--) {
      formattedString = numberString[i] + formattedString;
      counter++;
      if (counter % 3 == 0 && i != 0) {
        formattedString = '.' + formattedString;
      }
    }
    
    return 'Rp$formattedString';
  }
}

// Extension untuk Product
extension ProductExtensions on Product {
  // Method untuk diskon
  Product withDiscount(double percentage) {
    var discountedPrice = price * (1 - percentage / 100);
    return Product(name, discountedPrice);
  }
  
  // Format untuk display
  String get displayInfo => '$name - ${price.toRupiah()}';
  
  // Check jika mahal
  bool get isExpensive => price > 1000000;
}

void main(){
  // Penggunaan
  var laptop = Product('Hp', 15000000);
  var discounted = laptop.withDiscount(10);
  print(discounted.displayInfo);  // 'Laptop - Rp13.500.000'
  print(laptop.isExpensive);      // true

  var keyboard = Product('Headset', 750000);
  print(keyboard.displayInfo);
  print(keyboard.isExpensive);
}

Ringkasan & Best Practices

📘 Ringkasan

  • Extension methods → tambah method ke tipe yang ada
  • Sintaks: extension Nama on Type
  • Support: method, getter, setter, operator
  • Generic extensions: on List<T>

⭐ Best Practices

  • Nama jelas: StringExtensions, DateTimeUtils
  • Group by type: Satu file per tipe utama
  • Avoid overuse: Jangan jadikan semua utility sebagai extension
  • Null safety: Handle nullable types dengan baik
  • Dokumentasi: Tambahkan komentar untuk public extensions
🚀 Intisari: Extension methods membuat kode lebih ekspresif, mengikuti prinsip open-closed, dan API makin ramah pengembang.

Extension Method Tidak Bisa Mengakses Member Private
//Nama: Muhammad bagas putra andeza
//NIM:251410003
//Kelas: Si2A
class Counter {
  int _value = 0; // private

  void increment() {
    _value++;
  }

  // Getter publik untuk mengakses nilai private
  int get value => _value;
}

extension CounterStats on Counter {
  int getValue() {
    // Mengakses melalui getter publik
    return value;
  }
}

void main() {
  var c = Counter();

  c.increment();

  print(c.getValue());
}
Penjelasan: * Member _value bersifat private karena diawali _. * Extension method tidak bisa langsung mengakses member private dari luar class. * Solusinya adalah membuat getter publik (value) di dalam class Counter. * Extension kemudian mengakses getter tersebut, bukan `_value` langsung.
